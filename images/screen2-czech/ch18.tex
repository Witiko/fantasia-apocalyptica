\chapter{Pád Babylóna}
\verse{1}{110}{Potom jsem viděl jiného anděla, jak s~velikou \motif{mocí} sestupuje z~nebe; a~země byla ozářena jeho slávou.}
\verse{2}{174}{Zvolal mohutným hlasem: „Padl, padl veliký \motif{Babylón} a~stal se doupětem démonů, skrýší všech nečistých duchů a~každého zlověstného a~nenáviděného ptáka;}
\verse{3}{188}{neboť vínem Božího hněvu pro smilství té nevěstky byly opojeny všecky národy, králové světa s~ní \motif{smilnili} a~bohatí kupci země bohatli z~její rozmařilosti a~přepychu.“}
\verse{4}{140}{A slyšel jsem jiný hlas z~\motif{nebe}: „Vyjdi, lide můj, z~toho města, nemějte účast na jeho hříších, aby vás nestihly jeho pohromy.}
\verse{5}{78}{Neboť jeho hříchy se navršily až k~nebi a~Bůh nezapomněl na jeho viny.}
\verse{6}{141}{Odplaťte po zásluze té nevěstce, dvojnásob jí odplaťte za její činy! V~poháru, který připravovala, připravte pro ni dvojnásob;}
\verse{7}{160}{kolik si užila slávy a~hýření, tolik jí dejte teď \motif{trýzně} a~žalu. Protože si namlouvá: Trůním jako královna, vdovou nejsem a~neokusím smutku –}
\verse{8}{145}{proto v~jediném dni dopadnou na ni pohromy: smrt i~žal a~hlad, a~bude zničena ohněm. Neboť mocný je Hospodin, \motif{Bůh}, který ji odsoudil.“}
\verse{9}{128}{Pak budou naříkat nad ní a~bědovat \motif{králové} světa, kteří s~ní smilnili a~hýřili, až uvidí dým hořícího města;}
\verse{10}{184}{z hrůzy nad jeho zkázou neodváží se přiblížit a~budou naříkat: „Běda, běda, ty veliký \motif{Babylóne}, město tak mocné, jak v~jedinou hodinu byl nad tebou vykonán soud!“}
\verse{11}{99}{A bohatí kupci země naříkají a~bědují nad ním, protože už nikdo nekoupí jejich zboží:}
\verse{12}{203}{náklady \coloredmotif{gold}{zlata} a~stříbra, drahokamů a~perel, kmentu a~purpuru, hedvábí a~šarlatu, cedrového dřeva a~předmětů ze slonoviny a~předmětů ze vzácných dřev, náklady mědi, železa a~mramoru,}
\verse{13}{151}{skořice a~indického koření, voňavek, mastí a~vykuřovadel, vína a~oleje, mouky a~obilí, dobytčat a~ovcí, koní a~vozů a~otroků a~zajatců.}
\verse{14}{109}{Plody, které jsi dychtivě sklízelo, jsou pryč, všechen lesk a~nádhera zašly a~není po nich památky.}
\verse{15}{148}{Kupci, kteří s~tím vším obchodovali a~z~toho města bohatli, z~hrůzy nad jeho zkázou neodváží se přiblížit a~budou \motif{plakat} a~naříkat:}
\verse{16}{138}{„Běda, běda, tak veliké město, které se oblékalo do kmentu, purpuru a~šarlatu, které se zdobilo \coloredmotif{gold}{zlatem}, drahokamy a~perlami –}
\verse{17}{170}{a v~jedinou hodinu je zničeno takové bohatství!“ A~všichni velitelé lodí, dopravci zboží, námořníci a~kdokoli se živí plavbou, neodváží se přiblížit}
\verse{18}{104}{a budou volat, až uvidí \motif{dým} toho hořícího města: „Co se mohlo rovnat tomu velikému městu!“}
\verse{19}{206}{A sypou si prach na hlavu a~v~\motif{pláči} a~nářku křičí: „Běda, běda, tak veliké město, na jehož blahobytu zbohatli všichni majitelé \motif{námořních} lodí – a~v~jedinou hodinu bylo zpustošeno!“}
\verse{20}{126}{Raduj se, nebe a~svatí, \motif{apoštolové} i~proroci, protože Bůh vynesl nad tím městem rozsudek, jaký ono vyneslo nad vámi.}
\verse{21}{185}{A jeden silný \motif{anděl} pozvedl balvan, těžký jako mlýnský kámen, vrhl jej do moře a~zvolal: „Tak rázem bude svržen Babylón, to veliké město, a~nebude po něm ani památky.}
\verse{22}{167}{Nikdy už v~tobě nezazní \motif{hudba}, neozve se harfa, píšťala ani trubka, nikdy už v~tobě nebude řemeslník ani umělec, nikdo už v~tobě neuslyší zvuk mlýnů,}
\verse{23}{182}{nikdy už v~tobě nezasvitne světlo \motif{lampy}, nikdo už v~tobě nezaslechne hlas ženicha a~nevěsty. Tvoji obchodníci vládli světem a~tvé čarovné nápoje mámily celé národy;}
\verse{24}{84}{ve tvých zdech tekla \coloredmotif{blood}{krev} proroků a~svatých a~všech zavražděných na zemi.“}
