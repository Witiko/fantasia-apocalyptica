\chapter{Dravé šelmy vládnou zemí}
\verse{1}{174}{Tu jsem viděl, jak se z~moře vynořila dravá \motif{šelma} o~deseti rozích a~sedmi hlavách; na těch rozích deset královských \motif{korun} a~na hlavách jména urážející Boha.}
\verse{2}{161}{Ta šelma, kterou jsem viděl, byla jako levhart, její nohy jako tlapy medvěda a~její tlama jako tlama lví. A~drak jí dal svou \motif{sílu} i~trůn i~velikou moc.}
\verse{3}{125}{Jedna z~jejích hlav vypadala jako smrtelně raněná, ale ta rána se \motif{zahojila}. A~celá země v~obdivu šla za tou šelmou;}
\verse{4}{176}{klekali před drakem, protože dal té šelmě svou \motif{moc}, a~klekali také před šelmou a~volali: „Kdo se může rovnat té dravé šelmě, kdo se odváží s~ní bojovat?“}
\verse{5}{89}{A bylo jí dáno, aby mluvila pyšně a~rouhavě a~měla \motif{moc} po čtyřicet dva měsíce.}
\verse{6}{104}{A tak otevřela ústa a~\motif{rouhala se} Bohu, jeho jménu i~jeho příbytku, všem, kdo přebývají v~nebi.}
\verse{7}{137}{A bylo jí dáno, aby vedla válku proti svatým a~aby nad nimi zvítězila. Dostala \motif{moc} nad každým kmenem, národem, jazykem i~\motif{rasou};}
\verse{8}{153}{budou před ní klekat všichni obyvatelé země, jejichž jména nejsou od stvoření světa zapsána v~\motif{knize} života, v~\motif{knize} toho zabitého \motif{Beránka}.}
\verse{9}{21}{Kdo má \motif{uši}, slyš!}
\verse{10}{146}{Kdo má jít do zajetí, půjde do zajetí. Kdo má \motif{zemřít} mečem, musí mečem \motif{zemřít}. Teď musí Boží lid osvědčit trpělivost a~víru.}
\verse{11}{107}{Vtom jsem viděl jinou šelmu, jak vyvstala ze země: měla dva \motif{rohy} jako beránek, ale mluvila jako drak.}
\verse{12}{170}{Z pověření té první šelmy vykonává veškerou její moc. Nutí zemi a~její obyvatele, aby klekali před první šelmou, které se \motif{zahojila} její smrtelná rána.}
\verse{13}{93}{A činí veliká \motif{znamení}, dokonce i~oheň z~nebe nechá před zraky lidí sestoupit na zem.}
\verse{14}{213}{Bylo jí dáno dělat znamení ke cti první \motif{šelmy} a~svádět jimi obyvatele země; rozkazuje obyvatelům země, aby postavili sochu té \motif{šelmě}, která byla smrtelně zraněna mečem, a~přece zůstala naživu.}
\verse{15}{144}{Je jí dáno, aby do sochy té šelmy vdechla život, takže ta socha mluvila a~vydala rozkaz, že \motif{zemřou} všichni, kdo před ní nepokleknou.}
\verse{16}{119}{A nutí všechny, malé i~veliké, \motif{bohaté} i~chudé, svobodné i~otroky, aby měli na pravé ruce nebo na čele \motif{cejch},}
\verse{17}{104}{aby nemohl kupovat ani prodávat, kdo není \motif{označen} jménem té šelmy nebo číslicí jejího jména.}
\verse{18}{146}{To je třeba \motif{pochopit}: kdo má rozum, ať sečte číslice té šelmy. To číslo označuje člověka, a~je to číslo \motif{šest set šedesát šest}.}
