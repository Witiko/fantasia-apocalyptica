\chapter{Spoutání satana, tisíciletá říše, konečné vítězství a~soud}
\verse{1}{101}{Tu jsem viděl, jak z~nebe sestupuje anděl, který má v~ruce klíč od \motif{propasti} a~veliký řetěz.}
\verse{2}{62}{Zmocnil se draka, toho dávného hada, toho ďábla a~\motif{satana},}
\verse{3}{195}{na tisíc let jej spoutal, uvrhl do \motif{propasti}, uzamkl ji a~zapečetil, aby již nemohl klamat národy, dokud se nedovrší těch \motif{tisíc let}. Potom musí být ještě na krátký čas propuštěn.}
\verse{4a}{263}{Viděl jsem \motif{trůny} a~na nich usedli ti, jimž byl svěřen soud. Spatřil jsem také ty, kdo byli sťati pro svědectví Ježíšovo a~pro slovo Boží, protože nepoklekli před dravou šelmou ani jejím obrazem a~nepřijali její znamení na čelo ani na ruku.}
\verse{4b}{181}{Nyní povstali k~životu a~ujali se \motif{vlády} s~\motif{Kristem} na tisíc let. (Ostatní mrtví však nepovstanou k~životu, dokud se těch tisíc let nedovrší.) To je první vzkříšení.}
\verse{6}{218}{\motif{Blahoslavený} a~svatý, kdo má podíl na prvním vzkříšení! Nad těmi druhá smrt nemá moci, nýbrž Bůh a~Kristus je učiní svými kněžími a~budou s~ním kralovat po tisíc let.}
\verse{7}{71}{Až se dovrší tisíc let, bude \motif{satan} propuštěn ze svého žaláře}
\verse{8}{138}{a vyjde, aby oklamal národy ve všech čtyřech úhlech světa, Góga i~Magóga. \motif{Shromáždí} je k~boji a~bude jich jako písku v~moři.}
\verse{9}{152}{Viděl jsem, jak vystoupili po celé šíři země a~obklíčili tábor svatých a~\motif{město}, které miluje Bůh. Ale sestoupil oheň z~nebe a~pohltil je.}
\verse{10}{160}{Jejich svůdce ďábel byl uvržen do jezera, kde hoří síra a~kde je již dravá šelma i~falešný prorok. A~budou \motif{trýzněni} dnem i~nocí na věky věků.}
\verse{11}{137}{A viděl jsem veliký bělostný \motif{trůn} a~toho, kdo na něm seděl; před jeho pohledem zmizela země i~nebe a~už pro ně nebylo místa.}
\verse{12}{210}{Viděl jsem mrtvé, mocné i~prosté, jak stojí před trůnem, a~byly otevřeny knihy. Ještě jedna \motif{kniha} byla \motif{otevřena}, kniha života. A~mrtví byli souzeni podle svých činů zapsaných v~těch knihách.}
\verse{13}{114}{\motif{Moře} vydalo své mrtvé, i~smrt a~její říše vydaly své mrtvé, a~všichni byli souzeni podle svých činů.}
\verse{14}{99}{Pak \motif{smrt} i~\motif{její říše} byly uvrženy do hořícího jezera. To je druhá smrt: hořící jezero.}
\verse{15}{72}{A kdo nebyl zapsán v~\motif{knize} života, byl uvržen do hořícího jezera.}
