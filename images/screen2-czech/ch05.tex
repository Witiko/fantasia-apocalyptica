\chapter{Sedmkrát zapečetěnou knihu přijal Beránek}
\verse{1}{112}{A v~pravici toho, který sedí na trůnu, spatřil jsem \motif{knihu} úplně popsanou, zapečetěnou sedmi \motif{pečetěmi}.}
\verse{2}{133}{Tu jsem uviděl mocného anděla, který vyhlásil velikým hlasem: „Kdo je hoden \motif{otevřít} tu knihu a~rozlomit její pečetě?“}
\verse{3}{92}{Ale nikdo na nebi ani na zemi ani pod zemí nemohl tu knihu \motif{otevřít} a~podívat se do ní.}
\verse{4}{100}{Velmi jsem \motif{plakal}, že se nenašel nikdo, kdo by byl hoden tu knihu otevřít a~podívat se do ní.}
\verse{5}{149}{Ale jeden ze starců mi řekl: „\motif{Neplač}. Hle, zvítězil lev z~pokolení Judova, potomek \motif{Davidův}; on otevře tu knihu sedmkrát zapečetěnou.“}
\verse{6}{209}{Vtom jsem spatřil, že uprostřed mezi trůnem a~těmi čtyřmi bytostmi a~starci stojí \motif{Beránek}, ten obětovaný; měl sedm rohů a~sedm očí, což je sedmero \motif{duchů} Božích vyslaných do celého světa.}
\verse{7}{76}{Přistoupil k~tomu, který sedí na trůnu, a~přijal knihu z~jeho pravice.}
\verse{8}{198}{A když tu knihu uchopil, čtyři bytosti a~čtyřiadvacet \motif{starců} padlo na kolena před Beránkem; každý měl loutnu a~\coloredmotif{gold}{zlatou} nádobu naplněnou vůní kadidla, což jsou modlitby Božího lidu.}
\verse{9}{190}{A zpívali novou píseň: „Jsi \motif{hoden} přijmout tu knihu a~rozlomit její pečetě, protože jsi byl obětován, svou \coloredmotif{blood}{krví} jsi Bohu vykoupil lidi ze všech kmenů, jazyků, národů a~ras}
\verse{10}{84}{a učinil je \motif{královským} kněžstvem našeho Boha; a~ujmou se vlády nad zemí.“}
\verse{11}{128}{A viděl jsem, jak kolem trůnu a~těch bytostí i~starců stojí množství \motif{andělů} – bylo jich na tisíce a~na statisíce;}
\verse{12}{158}{slyšel jsem je mocným hlasem volat: „Hoden jest \motif{Beránek}, ten obětovaný, přijmout moc, bohatství, moudrost, sílu, poctu, slávu i~dobrořečení.“}
\verse{13}{207}{A všechno stvoření na nebi, na zemi, pod zemí i~v~moři, všecko, co v~nich jest, slyšel jsem volat: „Tomu, jenž sedí na trůnu, i~\motif{Beránkovi} dobrořečení, čest, sláva i~moc na věky věků!“}
\verse{14}{79}{A ty čtyři bytosti řekly: „\motif{Amen};“ starci padli na kolena a~klaněli se.}
