\chapter{Dopisy do Sard, Filadelfie a~Laodikeje}
\verse{1}{172}{Andělu \motif{církve} v~Sardách piš: Toto praví ten, který má sedmero \motif{duchů} Božích a~sedmero hvězd: „Vím o~tvých skutcích – podle jména jsi živ, ale jsi mrtev.}
\verse{2}{135}{\motif{Probuď se} a~posilni to, co ještě zůstává a~je už na vymření! Neboť shledávám, že tvým skutkům mnoho chybí před Bohem;}
\verse{3}{200}{rozpomeň se tedy, jak jsi mé slovo přijal a~slyšel, zachovávej je a~\motif{čiň pokání}. Nebudeš-li bdít, přijdu tak, jako přichází zloděj, a~nebudeš vědět, v~kterou hodinu na tebe přijdu.}
\verse{4}{147}{Přece však máš v~Sardách několik osob, které svůj šat nepotřísnily; ti budou chodit se mnou v~\motif{bílém} rouchu, protože jsou toho hodni.}
\verse{5}{157}{Kdo zvítězí, bude oděn \motif{bělostným rouchem} a~jeho jméno nevymažu z~knihy života, nýbrž přiznám se k~němu před svým Otcem a~před jeho \motif{anděly}.}
\verse{6}{50}{Kdo má \motif{uši}, slyš, co Duch praví církvím.“}
\verse{7}{175}{Andělu \motif{církve} ve Filadelfii piš: Toto prohlašuje ten Svatý a~Pravý, který má klíč \motif{Davidův}; když on otvírá, nikdo nezavře, a~když on zavírá, nikdo neotevře:}
\verse{8}{184}{„Vím o~tvých skutcích. Hle, otevřel jsem před tebou dveře, a~nikdo je nemůže zavřít. Neboť ačkoli máš nepatrnou \motif{moc}, zachoval jsi mé slovo a~mé jméno jsi nezapřel.}
\verse{9}{202}{Hle, dávám do tvých rukou ty, kdo jsou ze synagógy satanovy; říkají si židé, a~nejsou, ale lžou. Hle, způsobím, že přijdou a~padnou ti k~\motif{nohám}; a~poznají, že já jsem si tě zamiloval.}
\verse{10}{143}{Protože jsi zachoval mé slovo a~vytrval, zachovám tě i~já v~hodině zkoušky, která \motif{přijde} na celý svět a~prověří obyvatele země.}
\verse{11}{86}{Přijdu brzy; \motif{drž se} toho, co máš, aby tě nikdo nepřipravil o~vavřín vítěze.}
\verse{12}{225}{Kdo zvítězí, toho učiním sloupem v~chrámě svého Boha a~chrám již neopustí; napíšu na něj jméno svého Boha a~jméno jeho města, nového \motif{Jeruzaléma}, který sestupuje z~nebe od mého Boha, i~jméno své nové.}
\verse{13}{50}{Kdo má \motif{uši}, slyš, co Duch praví církvím.“}
\verse{14}{133}{Andělu \motif{církve} v~Laodikeji piš: Toto praví ten, jehož jméno jest \motif{Amen}, svědek věrný a~pravý, počátek stvoření Božího:}
\verse{15}{92}{„Vím o~tvých skutcích; nejsi studený ani horký. Kéž bys byl studený anebo horký!}
\verse{16}{74}{Ale že jsi vlažný, a~nejsi horký ani studený, \motif{nesnesu} tě v~ústech.}
\verse{17}{129}{Vždyť říkáš: Jsem \motif{bohat}, mám všecko a~nic už nepotřebuji! A~nevíš, že jsi \motif{ubohý}, bědný a~nuzný, slepý a~nahý.}
\verse{18}{180}{Radím ti, abys u~mne nakoupil \coloredmotif{gold}{zlata} ohněm přečištěného, a~tak zbohatl; a~bílý šat, aby ses oblékl a~nebylo vidět tvou nahotu; a~mast k~potření očí, abys prohlédl.}
\verse{19}{78}{Já kárám a~trestám ty, které miluji; vzpamatuj se tedy a~\motif{čiň pokání}.}
\verse{20}{131}{Hle, stojím přede dveřmi a~\motif{tluču}; zaslechne-li kdo můj hlas a~otevře mi, vejdu k~němu a~budu s~ním večeřet a~on se mnou.}
\verse{21}{113}{Kdo \motif{zvítězí}, tomu dám usednout se mnou na trůn, tak jako já jsem zvítězil a~usedl s~Otcem na jeho trůn.}
\verse{22}{50}{Kdo má \motif{uši}, slyš, co Duch praví církvím.“}
