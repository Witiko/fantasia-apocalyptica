\chapter{Vstupní slovo a~povolání~Jana}
\verse{1}{167}{Zjevení, které Bůh dal \motif{Ježíši Kristu}, aby ukázal svým služebníkům, co se má brzo stát; naznačil to prostřednictvím anděla svému služebníku \motif{Janovi}.}
\verse{2}{78}{Ten dosvědčil Boží slovo a~svědectví \motif{Ježíše Krista}, vše, co viděl.}
\verse{3}{140}{Blaze tomu, kdo předčítá slova tohoto proroctví, a~blaze těm, kdo \motif{slyší} a~zachovávají, co je tu napsáno, neboť čas je blízko.}
\verse{4}{141}{\motif{Jan} sedmi církvím v~Asii: \motif{Milost} vám a~pokoj od toho, který jest a~který byl a~který přichází, i~od sedmi duchů před jeho trůnem}
\verse{5}{158}{a od \motif{Ježíše Krista}, věrného svědka, prvorozeného z~mrtvých a~vládce \motif{králů} země. Jemu, jenž nás miluje a~svou \coloredmotif{blood}{krví} nás zprostil hříchů}
\verse{6}{95}{a učinil nás královským kněžstvem Boha, svého Otce – jemu sláva i~\motif{moc} navěky. \motif{Amen}.}
\verse{7}{149}{Hle, přichází v~\motif{oblacích}! Uzří ho každé oko, i~ti, kdo ho probodli, a~budou kvůli němu naříkat všechna pokolení země. Tak jest, amen.}
\verse{8}{107}{Já jsem \motif{Alfa} i~\motif{Omega}, praví Pán Bůh, ten, který jest a~který byl a~který přichází, Všemohoucí.}
\verse{9}{189}{Já, \motif{Jan}, váš bratr, který má s~vámi účast na Ježíšově soužení i~kralování a~vytrvalosti, dostal jsem se pro slovo Boží a~svědectví Ježíšovo na \motif{ostrov} jménem Patmos.}
\verse{10}{104}{Ocitl jsem se ve vytržení ducha v~den Páně, a~uslyšel jsem za sebou mocný hlas jako zvuk \motif{polnice}:}
\verse{11}{144}{„Co vidíš, napiš do \motif{knihy} a~pošli sedmi církvím: do Efezu, do Smyrny, do Pergama, do Thyatir, do Sard, do Filadelfie a~do Laodikeje.“}
\verse{12}{120}{Obrátil jsem se, abych viděl, kdo se mnou mluví. A~když jsem se obrátil, spatřil jsem sedm \coloredmotif{gold}{zlatých} \motif{svícnů};}
\verse{13}{109}{uprostřed těch svícnů někdo jako Syn \motif{člověka}, oděný řízou až na zem, a~na prsou \coloredmotif{gold}{zlatý} pás.}
\verse{14}{83}{Jeho hlava a~vlasy bělostné jako sněhobílá vlna, jeho \motif{oči} jako plamen ohně;}
\verse{15}{82}{jeho \motif{nohy} podobné kovu přetavenému ve výhni a~jeho hlas jako hukot příboje.}
\verse{16}{136}{V pravici držel sedm \motif{hvězd} a~z~jeho úst vycházel ostrý dvousečný \motif{meč}; jeho vzhled jako když \motif{slunce} září v~plné své síle.}
\verse{17}{147}{Když jsem ho spatřil, padl jsem k~jeho nohám jako mrtvý; ale on vložil na mne svou pravici a~řekl: „\motif{Neboj se}. Já jsem první i~poslední,}
\verse{18}{96}{ten živý; byl jsem mrtev – a~hle, \motif{živ} jsem na věky věků. Mám klíče od smrti i~hrobu.}
\verse{19}{73}{\motif{Napiš} tedy, co jsi viděl – to, co jest, i~to, co se má stát potom.}
\verse{20}{180}{Tajemství těch sedmi \motif{hvězd}, které jsi viděl v~mé pravici, i~těch sedmi \coloredmotif{gold}{zlatých} svícnů: Sedm hvězd jsou andělé sedmi církví, a~sedm \motif{svícnů} je sedm církví.“}
