\chapter{Záchrana vyvolených}
\verse{1}{188}{Potom jsem viděl, jak se čtyři \motif{andělé} postavili do čtyř úhlů země a~bránili všem čtyřem větrům, aby žádný z~nich nevál na zemi ani na moře ani na jakékoli stromoví.}
\verse{2}{175}{A hle, jiný anděl vystupoval od východu slunce; v~ruce držel \motif{pečetidlo} živého Boha a~mocným hlasem volal na ty čtyři anděly, jimž bylo dáno škodit zemi i~moři:}
\verse{3}{108}{„Neškoďte zemi, moři ani stromoví, dokud \motif{neoznačíme} služebníky našeho Boha na jejich čelech!“}
\verse{4}{110}{Pak jsem slyšel počet označených: \motif{sto čtyřiačtyřicet tisíc} označených ze všech pokolení Izraele:}
\verse{5}{104}{z pokolení \motif{Juda} dvanáct tisíc, z~pokolení Rúben dvanáct tisíc, z~pokolení Gád dvanáct tisíc,}
\verse{6}{110}{z pokolení Ašer dvanáct tisíc, z~pokolení Neftalím dvanáct tisíc, z~pokolení Manase dvanáct tisíc,}
\verse{7}{109}{z pokolení Šimeón dvanáct tisíc, z~pokolení Levi dvanáct tisíc, z~pokolení Isachar dvanáct tisíc,}
\verse{8}{125}{z pokolení Zabulón dvanáct tisíc, z~pokolení Josef dvanáct tisíc, z~pokolení Benjamín dvanáct tisíc označených.}
\verse{9}{235}{Potom jsem viděl, hle, tak veliký zástup, že by ho nikdo nedokázal sečíst, ze všech \motif{ras}, kmenů, národů a~jazyků, jak stojí před trůnem a~před tváří Beránkovou, oblečeni v~bílé roucho, palmové ratolesti v~rukou.}
\verse{10}{99}{A volali velikým hlasem: „Díky \motif{Spasiteli}, Bohu našemu, sedícímu na trůnu, a~\motif{Beránkovi}.“}
\verse{11}{124}{A všichni andělé se postavili kolem trůnu, kolem \motif{starců} i~těch čtyř \motif{bytostí} a~padli před trůnem tváří k~zemi,}
\verse{12}{139}{klaněli se Bohu a~volali: „\motif{Amen}! Dobrořečení i~sláva a~moudrost, \motif{díky} a~čest i~moc a~síla Bohu našemu na věky věků. \motif{Amen}!“}
\verse{13}{93}{Jeden z~těch starců na mne promluvil: „Kdo jsou a~odkud přišli ti v~\motif{bílém} rouchu?“}
\verse{14}{168}{Řekl jsem: „Pane můj, ty to víš!“ A~on mi řekl: „To jsou ti, kteří přišli z~velikého soužení a~vyprali svá roucha a~vybílili je v~\coloredmotif{blood}{krvi} \motif{Beránkově}.}
\verse{15}{129}{Proto jsou před trůnem Božím a~slouží mu v~jeho \motif{chrámě} dnem i~nocí; a~ten, který sedí na trůnu, bude jim záštitou.}
\verse{16}{82}{Již nebudou hladovět ani žíznit, ani slunce nebo jiný žár jim neublíží,}
\verse{17}{136}{neboť \motif{Beránek}, který je před trůnem, je bude pást a~povede je k~pramenům vod života. A~Bůh jim setře každou slzu z~\motif{očí}.“}
