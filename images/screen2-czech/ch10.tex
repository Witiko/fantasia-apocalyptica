\chapter{Nové povolání k~proroctví}
\verse{1}{165}{Tu jsem viděl dalšího mocného anděla, jak sestupuje z~nebe, zahalen v~oblak. Nad jeho hlavou byla duha, jeho tvář byla jako \motif{slunce}, nohy jako ohnivé sloupy;}
\verse{2}{88}{v ruce měl otevřenou \motif{knížečku}. Pravou nohou se postavil na moře, levou na pevninu}
\verse{3}{101}{a vykřikl mocným hlasem, jako když zařve lev. Na jeho výkřik odpovědělo sedmero \motif{zahřmění}.}
\verse{4}{172}{Jakmile doznělo to sedmero zahřmění, chtěl jsem to \motif{zapsat}; ale uslyšel jsem hlas z~nebe: „Zachovej v~tajnosti, co se ozvalo v~sedmeru zahřmění, a~nic nepiš!“}
\verse{5}{89}{Potom anděl, kterého jsem viděl stát na \motif{moři} i~na zemi, pozvedl svou pravici k~nebi}
\verse{6}{143}{a přísahou při tom, který je živ na věky věků, který stvořil nebesa, \motif{moře} a~všecko, co je v~nich, potvrdil, že lhůta je u~konce;}
\verse{7}{130}{ve dnech, kdy zazní polnice sedmého anděla, naplní se Boží tajemství, jak je \motif{Bůh} oznámil svým služebníkům prorokům.}
\verse{8}{146}{Hlas, který jsem slyšel z~nebe, opět ke mně promluvil: „Jdi, vezmi tu otevřenou \motif{knihu} z~ruky anděla, který stojí na \motif{moři} i~na zemi.“}
\verse{9}{180}{Přistoupil jsem tedy k~tomu andělu a~požádal ho, aby mi tu \motif{knihu} dal. Řekl mi: „Vezmi ji a~sněz; v~žaludku ti zhořkne, ačkoli v~tvých ústech bude sladká jako med.“}
\verse{10}{137}{Vzal jsem tu \motif{knihu} z~ruky anděla a~snědl ji; v~mých ústech byla sladká jako med, ale když jsem ji pozřel, zhořkla mi v~žaludku.}
\verse{11}{112}{Tehdy jsem slyšel: „Je třeba, abys znovu prorokoval proti mnoha národům, \motif{kmenům}, jazykům i~králům.“}
