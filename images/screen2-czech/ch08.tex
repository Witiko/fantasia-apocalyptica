\chapter{Sedmá pečeť a~andělé s~polnicemi}
\verse{1}{92}{A když Beránek \motif{rozlomil} pečeť sedmou, nastalo na nebi mlčení téměř na půl hodiny.}
\verse{2}{88}{Potom jsem viděl, jak sedmi \motif{andělům}, stojícím před Bohem, bylo dáno sedm \motif{polnic}.}
\verse{3}{181}{Jiný anděl předstoupil se \coloredmotif{gold}{zlatou} kadidelnicí před oltář; bylo mu dáno množství kadidla, aby je s~modlitbami všech posvěcených položil na \coloredmotif{gold}{zlatý} oltář před \motif{trůnem}.}
\verse{4}{94}{A vystoupil \motif{dým} kadidla spolu s~modlitbami posvěcených z~ruky anděla před Boží tvář.}
\verse{5}{144}{Tu vzal anděl kadidelnici, nahrnul do ní oheň z~oltáře a~vrhl ji dolů na zem; a~nastalo burácení, hřímání, blesky a~\motif{zemětřesení}.}
\verse{6}{65}{A sedm andělů s~polnicemi se připravilo, aby začali troubit.}
\verse{7}{163}{\motif{Zatroubil} první anděl: Nastalo krupobití a~na zem začal padat oheň smíšený s~\coloredmotif{blood}{krví}. Třetina země, třetina stromoví a~veškerá zeleň byla sežehnuta.}
\verse{8}{123}{\motif{Zatroubil} druhý anděl; a~jakoby mohutná hora hořící ohněm byla vržena do moře. Třetina moře se obrátila v~\coloredmotif{blood}{krev}}
\verse{9}{70}{a zahynula třetina \motif{mořských} tvorů a~byla zničena třetina lodí.}
\verse{10}{131}{\motif{Zatroubil} třetí anděl, a~zřítila se z~nebe veliká hvězda hořící jako pochodeň, padla na třetinu řek a~na prameny vod.}
\verse{11}{135}{(Jméno té \motif{hvězdy} je Pelyněk.) Třetina vod se změnila v~pelyněk a~množství lidí umřelo z~těch vod, protože byly otráveny.}
\verse{12}{170}{\motif{Zatroubil} čtvrtý anděl, a~byla zasažena třetina slunce, třetina měsíce a~třetina hvězd, takže ze třetiny potemněly, a~den i~noc byly o~třetinu temnější.}
\verse{13}{200}{A viděl jsem a~slyšel, jak středem nebeské klenby letí \motif{orel} a~volá mocným hlasem: „Běda, běda, běda obyvatelům země, až zaznějí polnice tří andělů, kteří ještě netroubili!“}
