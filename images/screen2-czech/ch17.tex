\chapter{Soud nad nevěstkou}
\verse{1}{169}{Tu přišel jeden z~těch sedmi andělů, kteří měli sedm nádob, a~promluvil ke mně: „Pojď se mnou, ukážu ti soud nad velikou \motif{nevěstkou}, usazenou nad vodami,}
\verse{2}{99}{se kterou se spustili \motif{králové} světa a~vínem jejího smilství se opíjeli obyvatelé země.“}
\verse{3}{174}{Anděl mě odvedl ve vytržení ducha na poušť. Tu jsem spatřil ženu sedící na dravé \motif{šelmě} nachové barvy, plné rouhavých jmen, o~sedmi hlavách a~deseti rozích.}
\verse{4}{159}{Ta žena byla oděna purpurem a~šarlatem a~ozdobena \coloredmotif{gold}{zlatem}, drahokamy a~perlami; v~ruce držela \coloredmotif{gold}{zlatý} pohár, plný ohavností a~nečistoty svého smilství}
\verse{5}{134}{a na čele měla napsáno jméno – je v~něm tajemství: „\motif{Babylón} veliký, Matka všeho smilstva a~všech ohavností na zemi.“}
\verse{6}{119}{Viděl jsem tu ženu, zpitou \coloredmotif{blood}{krví} svatých a~\coloredmotif{blood}{krví} Ježíšových svědků. Velice jsem užasl, když jsem ji viděl.}
\verse{7}{135}{Ale anděl mi řekl: „Čemu se divíš? Já ti odhalím tajemství té ženy i~té sedmihlavé a~desetirohé \motif{šelmy}, která ji nese.}
\verse{8}{265}{Ta dravá šelma, kterou jsi viděl, byla a~není; vystoupí ještě z~propasti, ale půjde do záhuby. A~užasnou ti obyvatelé země, jejichž jméno není od založení světa zapsáno v~\motif{knize} života, až uvidí, že ta dravá šelma byla a~není, a~zase bude.}
\verse{9}{118}{Ať pochopí ten, komu je dána \motif{moudrost}. Sedm hlav je sedm pahorků, na nichž ta žena sedí, a~také sedm \motif{králů}:}
\verse{10}{105}{pět jich padlo, jeden kraluje, jeden ještě nepřišel. Až přijde, bude smět zůstat jen nakrátko.}
\verse{11}{109}{A ta dravá \motif{šelma}, která byla a~není, je osmý král, a~přece jeden z~těch sedmi; jde však do záhuby.}
\verse{12}{149}{Deset \motif{rohů}, které jsi viděl, je deset králů, kteří se ještě vlády neujali, ale v~jedinou hodinu přijmou královskou moc spolu se šelmou.}
\verse{13}{72}{Budou zajedno ve svých úmyslech a~svou \motif{sílu} i~moc dají té šelmě.}
\verse{14}{161}{Ti budou bojovat s~\motif{Beránkem}, ale \motif{Beránek} je přemůže, protože je Pán pánů a~Král králů; ti, kdo jsou s~ním, jsou povolaní a~vyvolení a~věrní.“}
\verse{15}{109}{A řekl mi: „Vody, které jsi viděl, nad nimiž ta \motif{nevěstka} sedí, to jsou národy, davy, rasy a~jazyky.}
\verse{16}{161}{A těch deset \motif{rohů}, které jsi viděl, i~ta šelma pojmou nevěstku v~nenávist, oberou ji o~všecko až do naha a~budou rvát její tělo a~spálí ji ohněm.}
\verse{17}{167}{Neboť Bůh jim vložil do srdce, aby provedli jeho záměr, řídili se jedním úmyslem a~odevzdali \motif{šelmě} svou královskou moc, dokud se nedokoná, co Bůh řekl.}
\verse{18}{80}{Ta žena, kterou jsi viděl, je veliké \motif{město}, panující nad \motif{králi} země.“}
