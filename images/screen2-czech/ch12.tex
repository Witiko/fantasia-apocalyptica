\chapter{Poražený nepřítel}
\verse{1}{128}{A ukázalo se veliké znamení na nebi: Žena oděná \motif{sluncem}, s~měsícem pod nohama a~s~korunou dvanácti hvězd kolem hlavy.}
\verse{2}{79}{Ta žena byla těhotná a~křičela v~\motif{bolestech}, neboť přišla její hodina.}
\verse{3}{136}{Tu se ukázalo na nebi jiné znamení: Veliký ohnivý drak s~deseti \motif{rohy} a~sedmi hlavami, a~na každé hlavě měl královskou korunu.}
\verse{4}{133}{Ocasem smetl třetinu \motif{hvězd} z~nebe a~svrhl je na zem. A~drak se postavil před ženu, aby pohltil její dítě, jakmile se narodí.}
\verse{5}{128}{Ona porodila dítě, syna, který má železnou berlou \motif{pást} všechny národy; ale dítě bylo přeneseno k~Bohu a~jeho trůnu.}
\verse{6}{133}{\motif{Žena} pak uprchla na poušť, kde jí Bůh připravil útočiště, aby tam o~ni bylo postaráno po tisíc dvě stě šedesát dní.}
\verse{7}{71}{A strhla se \motif{bitva} na nebi: Michael a~jeho andělé se utkali s~\motif{drakem}.}
\verse{8}{87}{Drak i~jeho andělé bojovali, ale nezvítězili, a~nebylo již pro ně místa v~nebi.}
\verse{9}{131}{A veliký \motif{drak}, ten dávný had, zvaný ďábel a~satan, který sváděl celý svět, byl svržen na zem a~s~ním i~jeho \motif{andělé}.}
\verse{10}{212}{A slyšel jsem mocný hlas v~nebi: „Nyní přišlo spasení, moc a~\motif{království} našeho Boha i~vláda jeho Mesiáše; neboť byl svržen žalobce našich bratří, který je před Bohem osočoval dnem i~nocí.}
\verse{11}{128}{Oni nad ním \motif{zvítězili} pro \coloredmotif{blood}{krev} Beránkovu a~pro slovo svého svědectví. Nemilovali svůj život tak, aby se zalekli smrti.}
\verse{12}{178}{Proto jásejte, nebesa a~všichni, kdo v~nich přebýváte! Běda však zemi i~moři, neboť sestoupil k~vám \motif{ďábel}, plný zlosti, protože ví, jak málo času mu zbývá.“}
\verse{13}{93}{Když drak viděl, že je svržen na zem, začal pronásledovat ženu, která porodila \motif{syna}.}
\verse{14}{192}{Ale té ženě byla dána dvě mocná \motif{orlí} křídla, aby mohla uletět na poušť do svého útočiště, kde ukryta před hadem byla zachována při životě rok a~dva roky a~polovinu roku.}
\verse{15}{72}{A had za ní vychrlil ze chřtánu proud \motif{vody} jako řeku, aby ji smetl.}
\verse{16}{97}{Ale země přispěla ženě na pomoc, otevřela ústa a~pohltila tu řeku, kterou drak vychrlil.}
\verse{17}{173}{\motif{Drak} v~hněvu vůči té ženě rozpoutal válku proti ostatnímu jejímu potomstvu, proti těm, kdo zachovávají přikázání Boží a~drží se svědectví Ježíšova.}
\verse{18}{35}{A drak se postavil na \motif{břeh moře}.}
