\chapter{Hrůzy sedmi nádob}
\verse{1}{133}{A slyšel jsem mocný hlas ze svatyně, jak praví sedmi \motif{andělům}: „Jděte a~vylejte těch sedm nádob Božího hněvu na zem!“}
\verse{2}{157}{První anděl šel a~vylil svou \motif{nádobu} na zem: a~zlé, zhoubné vředy padly na lidi označené znamením dravé šelmy a~klekající před jejím obrazem.}
\verse{3}{113}{Druhý vylil svou \motif{nádobu} na moře: a~změnilo se v~\coloredmotif{blood}{krev} jako \coloredmotif{blood}{krev} zabitého a~všechno živé v~moři zahynulo.}
\verse{4}{73}{Třetí vylil svou \motif{nádobu} na řeky a~prameny vod: a~změnily se v~\coloredmotif{blood}{krev}.}
\verse{5}{156}{Tu jsem slyšel, jak praví anděl, který má moc nad vodami: „Spravedlivý jsi, Bože, svatý, který jsi a~kterýs byl, že jsi vynesl tento rozsudek:}
\verse{6}{92}{těm, kdo prolili \coloredmotif{blood}{krev} svatých a~proroků, dal jsi pít \coloredmotif{blood}{krev}; stalo se jim po zásluze!“}
\verse{7}{108}{A od oltáře jsem slyšel hlas: „Ano, Pane \motif{Bože} všemohoucí, pravé a~spravedlivé jsou tvé soudy.“}
\verse{8}{94}{Čtvrtý anděl vylil svou \motif{nádobu} na slunce: a~byla mu dána moc spalovat lidi svou výhní.}
\verse{9}{134}{Lidé hynuli nesmírným žárem a~\motif{proklínali} Boha, který má moc nad takovými pohromami; ale neobrátili se, aby mu vzdali čest.}
\verse{10}{87}{Pátý anděl vylil svou \motif{nádobu} na trůn šelmy: a~v~jejím království nastala tma,}
\verse{11}{120}{lidé se bolestí hryzali do rtů a~trýzněni vředy \motif{rouhali} se nebeskému Bohu, ale neodvrátili se od svých činů.}
\verse{12}{137}{Šestý anděl vylil svou \motif{nádobu} na velikou řeku Eufrat: a~její voda vyschla, aby byla připravena cesta králům od východu slunce.}
\verse{13}{126}{A hle, z~úst \motif{draka} i~z~úst dravé \motif{šelmy} a~z~úst \motif{lživého proroka} vystoupili tři nečistí duchové, podobní ropuchám.}
\verse{14}{172}{Jsou to duchové ďábelští, kteří činí zázračná znamení. Vyšli ke \motif{králům} celého světa, aby je \motif{shromáždili} k~boji v~rozhodující den všemohoucího Boha.}
\verse{15}{142}{„Hle, přicházím nečekaně jako zloděj! \motif{Blaze} tomu, kdo bdí a~střeží svůj šat, aby nechodil nahý a~nebylo vidět jeho nahotu!“}
\verse{16}{63}{\motif{Shromáždili} ty krále na místo zvané hebrejsky Harmagedon.}
\verse{17}{105}{Sedmý anděl vylil svou \motif{nádobu} do ovzduší a~z~chrámu od trůnu zazněl mocný hlas: „Stalo se!“}
\verse{18}{155}{A rozpoutaly se blesky, hřmění a~burácení, a~nastalo hrozné \motif{zemětřesení}, jaké nebylo, co je člověk na zemi; tak silné bylo to \motif{zemětřesení}.}
\verse{19}{179}{A to veliké \motif{město} se roztrhlo na tři části a~\motif{města} národů se zřítila. Bůh se rozpomenul na veliký Babylón a~dal mu pít z~poháru vína svého trestajícího hněvu.}
\verse{20}{55}{Všechny \motif{ostrovy} zmizely, po horách nezůstalo stopy,}
\verse{21}{133}{na lidi padaly z~nebe \motif{kroupy} těžké jako cent; a~lidé proklínali Boha za pohromu \motif{krupobití}, protože ta pohroma byla strašná.}
