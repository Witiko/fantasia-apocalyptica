\chapter{Ohlášení soudu}
\verse{1}{157}{A viděl jsem, hle, \motif{Beránek} stál na hoře Sión a~s~ním sto čtyřicet čtyři tisíce těch, kdo mají na čele napsáno jméno jeho i~jméno jeho Otce.}
\verse{2}{148}{A slyšel jsem hlas z~nebe jako hukot množství vod a~jako zvuk mocného hřmění; a~znělo to, jako když hudebníci rozezvučí své \motif{nástroje}.}
\verse{3}{211}{Zpívali novou píseň před trůnem, před těmi čtyřmi bytostmi a~před \motif{starci}. Nikdo nebyl schopen naučit se té písni, než těch sto čtyřicet čtyři tisíce z~obyvatel země, kteří byli vykoupeni.}
\verse{4}{169}{To jsou ti, kdo neporušili svou čistotu se ženami a~zůstali panici. Ti následují \motif{Beránka}, kamkoli jde. Ti jako první z~lidstva byli vykoupeni Bohu a~\motif{Beránkovi}.}
\verse{5}{54}{Z jejich úst nikdo neuslyšel lež; jsou bez úhony.}
\verse{6}{161}{Tu jsem viděl jiného \motif{anděla}, jak letí středem nebeské klenby, aby zvěstoval věčné evangelium obyvatelům země, každé rase, \motif{kmeni}, jazyku i~národu.}
\verse{7}{165}{Volal mocným hlasem: „\motif{Bojte se} Boha a~vzdejte mu čest, neboť nastala hodina jeho soudu; poklekněte před tím, kdo učinil nebe, zemi, moře i~prameny vod.“}
\verse{8}{154}{Za ním letěl druhý \motif{anděl} a~volal: „Padl, padl veliký Babylón, který opojil všechny národy svým smilstvím a~dal jim pít z~poháru hněvu.“}
\verse{9}{153}{Za nimi letěl třetí \motif{anděl} a~volal mocným hlasem: „Kdo kleká před šelmou a~před její sochou, kdo přijímá její \motif{cejch} na čelo či na ruku,}
\verse{10}{169}{bude pít \motif{víno} Božího rozhorlení, které Bůh nalévá neředěné do číše svého hněvu; a~bude mučen ohněm a~sírou před svatými anděly a~před Beránkem.}
\verse{11}{153}{A jeho muka \motif{neuhasnou} na věky věků a~dnem ani nocí nedojde pokoje ten, kdo kleká před šelmou a~jejím obrazem a~nechal si vtisknout její jméno.}
\verse{12}{104}{Zde se ukáže vytrvalost svatých, kteří zachovávají Boží přikázání a~\motif{věrnost} Ježíši.“}
\verse{13}{195}{A slyšel jsem hlas z~nebe: „Piš: Od této chvíle jsou \motif{blahoslaveni} mrtví, kteří umírají v~Pánu. Ano, praví Duch, ať odpočinou od svých prací, neboť jejich skutky jdou s~nimi.“}
\verse{14}{130}{A viděl jsem, hle, bílý \motif{oblak}, a~na \motif{oblaku} sedí někdo jako Syn člověka, na hlavě má korunu ze \coloredmotif{gold}{zlata} a~v~ruce ostrý srp.}
\verse{15}{190}{Vtom další anděl vyšel z~chrámu a~mocným hlasem zavolal na toho, který seděl na oblaku: „Pošli svůj \motif{srp} a~začni žeň, protože nastala hodina žně a~úroda země dozrála.“}
\verse{16}{78}{A ten, který seděl na oblaku, hodil svůj \motif{srp} na zem, a~země byla požata.}
\verse{17}{73}{A další anděl vyšel z~nebeského chrámu a~také on měl ostrý \motif{srp}.}
\verse{18}{206}{A jiný anděl vyšel od oltáře a~ten měl moc nad ohněm. Zavolal mocným hlasem na toho, který měl ostrý srp: „Pošli svůj ostrý \motif{srp} a~skliď hrozny na vinicích země, neboť víno dozrálo.“}
\verse{19}{105}{Tu hodil ten anděl svůj \motif{srp} na zem a~sklidil víno země a~uvrhl je do velikého lisu Božího hněvu.}
\verse{20}{126}{A v~\motif{lisu} za městem byly hrozny stlačeny a~vytékající \coloredmotif{blood}{krev} dosahovala až po uzdy koní na vzdálenost sto šedesát mil.}
