\chapter{Pozvání k~svatbě Beránkově a~Kristovo vítězství nad antikristem}
\verse{1}{126}{Potom jsem slyšel mocný hlas jakoby obrovského zástupu na nebi: „\motif{Haleluja}! Spasení, sláva i~moc patří Bohu našemu,}
\verse{2}{227}{protože pravé a~spravedlivé jsou jeho soudy! Vždyť odsoudil tu velikou \motif{nevěstku}, která přivedla zemi do záhuby svým smilstvím, spravedlivě ji potrestal za \coloredmotif{blood}{krev} svých služebníků, která lpí na jejích rukou.“}
\verse{3}{90}{A znovu volali: „\motif{Haleluja}! Na věky věků vystupuje \motif{dým} z~toho hořícího města.“}
\verse{4}{155}{Tehdy těch čtyřiadvacet \motif{starců} spolu se čtyřmi bytostmi padlo na kolena; klaněli se před Bohem sedícím na trůnu a~volali: „Amen, \motif{haleluja}.“}
\verse{5}{124}{A od trůnu zazněl hlas: „Zpívejte Bohu našemu, všichni jeho \motif{služebníci}, kteří se ho bojíte, malí i~velicí.“}
\verse{6}{156}{A slyšel jsem zpěv jakoby ohromného zástupu, jako hukot množství vod a~jako dunění hromu: „\motif{Haleluja}, ujal se vlády Pán Bůh náš všemohoucí.}
\verse{7}{102}{Radujme se a~jásejme a~vzdejme mu chválu; přišel den \motif{svatby} \motif{Beránkovy}, jeho choť se připravila}
\verse{8}{111}{a byl jí dán zářivě čistý \motif{kment}, aby se jím oděla.“ (Tím \motif{kmentem} jsou spravedlivé skutky svatých.)}
\verse{9}{129}{Tehdy mi řekl: „Piš: Blaze těm, kdo jsou pozváni na \motif{svatbu} \motif{Beránkovu}.“ A~řekl mi: „Toto jsou pravá slova Boží.“}
\verse{10}{249}{Tu jsem padl na kolena k~jeho nohám. Ale on mi řekl: „Střez se toho! Jsem jen služebník jako ty a~tvoji bratři, kteří vydávají svědectví \motif{Ježíšovi}. Před Bohem poklekni!“ Kdo vydává svědectví \motif{Ježíšovi}, má ducha proroctví.}
\verse{11}{149}{A viděl jsem nebesa otevřená, a~hle, bílý \motif{kůň}, a~na něm seděl ten, který má jméno Věrný a~Pravý, neboť soudí a~bojuje spravedlivě.}
\verse{12}{123}{Jeho \motif{oči} plamen ohně a~na hlavě množství královských korun; jeho jméno je napsáno a~nezná je nikdo než on sám.}
\verse{13}{69}{Má na sobě plášť zbrocený \coloredmotif{blood}{krví} a~jeho jméno je Slovo Boží.}
\verse{14}{90}{Za ním nebeská vojska na bílých \motif{koních}, oblečená do bělostného čistého kmentu.}
\verse{15}{169}{Z jeho úst vychází ostrý \motif{meč}, aby jím pobíjel národy; bude je \motif{pást} železnou berlou. On bude tlačit lis plný vína trestajícího hněvu Boha všemohoucího.}
\verse{16}{72}{Na plášti a~na boku má napsáno jméno: \motif{Král} \motif{králů} a~Pán pánů.}
\verse{17}{172}{A viděl jsem jednoho anděla, jak stojí na \motif{slunci} a~volá mocným hlasem na všechny ptáky letící středem nebes: „Pojďte, slétněte se k~veliké hostině Boží!}
\verse{18}{133}{Budete se sytit \motif{těly} králů a~vojevůdců a~bojovníků a~koní i~jezdců; \motif{těly} všech, pánů i~otroků, slabých i~mocných.“}
\verse{19}{130}{A viděl jsem dravou šelmu a~\motif{krále} země i~jejich vojska shromážděná, aby vytrhli do boje proti jezdci a~proti jeho vojsku.}
\verse{20}{256}{Ale \motif{šelma} byla zajata a~s~ní i~\motif{falešný prorok}, který k~její cti činil zázračná znamení a~svedl jimi ty, kdo přijali cejch šelmy a~klekali před jejím obrazem. Za živa byli šelma a~její prorok hozeni do ohnivého jezera hořícího sírou.}
\verse{21}{103}{Ostatní byli pobiti \motif{mečem} vycházejícím z~úst jezdce. A~všichni ptáci se nasytili jejich těly.}
