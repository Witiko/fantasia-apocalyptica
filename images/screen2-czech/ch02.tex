\chapter{Dopisy do Efezu, Smyrny, Pergama a~Thyatir}
\verse{1}{140}{Andělu \motif{církve} v~Efezu piš: Toto praví ten, který drží sedm hvězd ve své pravici, který se prochází mezi sedmi \coloredmotif{gold}{zlatými} \motif{svícny}:}
\verse{2}{209}{„Vím o~tvých skutcích, o~tvém úsilí i~tvé vytrvalosti; vím, že nemůžeš snést ty, kdo jsou zlí, a~vyzkoušel jsi ty, kdo se vydávají za \motif{apoštoly}, ale nejsou, a~shledal jsi, že jsou lháři.}
\verse{3}{68}{Máš \motif{vytrvalost} a~trpěl jsi pro mé jméno, a~nepodlehls únavě.}
\verse{4}{75}{Ale to mám proti tobě, že už nemáš takovou lásku jako na počátku.}
\verse{5}{152}{Rozpomeň se, odkud jsi klesl, \motif{navrať se} a~jednej jako dřív. Ne-li, přijdu na tebe a~pohnu tvým svícnem z~jeho místa, jestliže se neobrátíš.}
\verse{6}{77}{To však máš k~dobru, že nenávidíš skutky \motif{nikolaitů} stejně jako já.}
\verse{7}{118}{Kdo má uši, slyš, co Duch praví církvím: Tomu, kdo zvítězí, dám jíst ze \motif{stromu života} v~Božím ráji.“}
\verse{8}{97}{Andělu \motif{církve} ve Smyrně piš: Toto praví ten první i~poslední, který byl mrtev a~je živ:}
\verse{9}{155}{„Vím o~tvém soužení a~tvé chudobě, ale jsi \motif{bohat}; vím, jak tě urážejí ti, kdo si říkají židé, ale nejsou, nýbrž je to spolek satanův!}
\verse{10}{218}{Neboj se toho, co máš vytrpět. Hle, ďábel má některé z~vás uvrhnout do vězení, abyste prošli zkouškou, a~budete mít soužení po deset dní. Buď \motif{věrný} až na smrt, a~dám ti vítězný věnec života.}
\verse{11}{96}{Kdo má \motif{uši}, slyš, co Duch praví církvím: Kdo zvítězí, tomu druhá smrt neublíží.“}
\verse{12}{85}{Andělu \motif{církve} v~Pergamu piš: Toto praví ten, který má ostrý dvousečný \motif{meč}:}
\verse{13}{211}{„Vím, kde bydlíš: tam, kde je trůn satanův. Avšak pevně se držíš mého jména a~nezapřel jsi \motif{víru} ve mne ani ve dnech, kdy můj věrný svědek Antipas byl zabit mezi vámi, tam, kde bydlí satan.}
\verse{14}{170}{Jen to mám proti tobě, že u~sebe máš zastánce učení Balaámova. Jako on učil Baláka svádět syny Izraele, aby se účastnili modlářských hostin a~\motif{smilstva},}
\verse{15}{66}{tak i~ty máš některé, kteří zastávají učení \motif{nikolaitů}.}
\verse{16}{87}{Proto se obrať! Ne-li, brzo k~tobě přijdu a~budu s~nimi bojovat \motif{mečem} svých úst.}
\verse{17}{217}{Kdo má \motif{uši}, slyš, co Duch praví církvím: Tomu, kdo zvítězí, dám jíst ze skryté many; dám mu bílý kamének, a~na tom kaménku je napsáno nové jméno, které nezná nikdo než ten, kdo je dostává.“}
\verse{18}{122}{Andělu \motif{církve} v~Thyatirech piš: Toto praví Syn Boží, jenž má oči jako planoucí oheň a~\motif{nohy} jako zářivý kov:}
\verse{19}{115}{„Vím o~tvých skutcích, lásce a~\motif{víře}, službě a~vytrvalosti; vím, že tvých skutků je čím dál více.}
\verse{20}{183}{Ale to mám proti tobě, že trpíš ženu \motif{Jezábel}, která se vydává za prorokyni a~svým učením svádí moje služebníky ke smilstvu a~k~účasti na modlářských hostinách.}
\verse{21}{78}{Dal jsem jí čas k~pokání, ale ona se nechce \motif{odvrátit} od svého smilstva.}
\verse{22}{126}{Hle, sešlu na ni nemoc a~do velikého soužení uvrhnu ty, kdo s~ní cizoloží, jestliže se od jejích činů \motif{neodvrátí};}
\verse{23}{148}{a její děti \motif{zahubím}. Tu poznají všechny církve, že já vidím do nitra i~srdce člověka, a~každému z~vás odplatím podle vašich činů.}
\verse{24}{172}{Vám ostatním v~Thyatirech, kteří nepřijímáte toto učení, kteří jste nepoznali to, čemu říkají hlubiny \motif{satanovy}, pravím: Nevkládám na vás jiné břemeno.}
\verse{25}{56}{Jenom se pevně držte toho, co máte, dokud \motif{nepřijdu}.}
\verse{26}{84}{Kdo zvítězí a~setrvá v~mých skutcích až do konce, tomu dám moc nad národy:}
\verse{27}{74}{bude je \motif{pást} železnou berlou, jako hliněné nádobí je bude rozbíjet}
\verse{28}{98}{– tak jako já jsem tu moc přijal od svého Otce. Tomu, kdo zvítězí, dám \motif{hvězdu} jitřní.}
\verse{29}{51}{Kdo má \motif{uši}, slyš, co Duch praví církvím.“}
